### Database Schema (SQLite)
```sql
-- Key-Value Store
CREATE TABLE kv_store (
    namespace TEXT NOT NULL,
    key TEXT NOT NULL,
    value TEXT NOT NULL,
    expires_at INTEGER,
    created_at INTEGER DEFAULT (strftime('%s', 'now')),
    PRIMARY KEY (namespace, key)
);

-- Agent Logs
CREATE TABLE agent_logs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id TEXT NOT NULL,
    tool_name TEXT NOT NULL,
    action TEXT NOT NULL,
    input_data TEXT,
    output_data TEXT,
    timestamp INTEGER DEFAULT (strftime('%s', 'now')),
    tokens_used INTEGER DEFAULT 0,
    duration_ms INTEGER DEFAULT 0
);

-- Indexes for performance
CREATE INDEX idx_kv_namespace ON kv_store(namespace);
CREATE INDEX idx_kv_expires ON kv_store(expires_at);
CREATE INDEX idx_logs_session ON agent_logs(session_id);
CREATE INDEX idx_logs_timestamp ON agent_logs(timestamp);
```
