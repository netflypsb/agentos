// RBAC Implementation
class EnterpriseAuth {
  async checkPermission(userId: string, resource: string, action: string): Promise<boolean> {
    const userRoles = await this.getUserRoles(userId);
    const permissions = await this.getPermissionsForRoles(userRoles);
    return permissions.some(p => p.resource === resource && p.actions.includes(action));
  }
}

CREATE TABLE workspaces (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    organization_id TEXT NOT NULL,
    settings TEXT, -- JSON workspace settings
    resource_limits TEXT, -- JSON quotas and limits
    created_at INTEGER DEFAULT (strftime('%s', 'now'))
);

CREATE TABLE workspace_members (
    workspace_id TEXT NOT NULL,
    user_id TEXT NOT NULL,
    role TEXT NOT NULL, -- admin, member, viewer
    permissions TEXT, -- JSON additional permissions
    joined_at INTEGER DEFAULT (strftime('%s', 'now')),
    PRIMARY KEY (workspace_id, user_id)
);

-- Workspace-specific data isolation
CREATE TABLE workspace_kv_store (
    workspace_id TEXT NOT NULL,
    namespace TEXT NOT NULL,
    key TEXT NOT NULL,
    value TEXT NOT NULL,
    expires_at INTEGER,
    created_at INTEGER DEFAULT (strftime('%s', 'now')),
    PRIMARY KEY (workspace_id, namespace, key),
    FOREIGN KEY (workspace_id) REFERENCES workspaces(id)
);

class ThreatDetectionSystem {

    const threats = await this.threatIntelligence.identify(anomalies);
    const risk = await this.riskAssessor.assess(threats);
