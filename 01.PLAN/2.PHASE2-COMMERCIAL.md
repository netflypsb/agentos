## Technical Architecture (Phase 2)

### Enhanced Database Schema
```sql
-- License Management
CREATE TABLE licenses (
    id TEXT PRIMARY KEY,
    user_id TEXT NOT NULL,
    tier TEXT NOT NULL, -- free, pro, team, enterprise
    expires_at INTEGER,
    features TEXT, -- JSON array of enabled features
    created_at INTEGER DEFAULT (strftime('%s', 'now'))
);

-- Usage Tracking
CREATE TABLE usage_tracking (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id TEXT NOT NULL,
    tool_name TEXT NOT NULL,
    usage_count INTEGER DEFAULT 1,
    last_used INTEGER DEFAULT (strftime('%s', 'now')),
    date TEXT NOT NULL -- YYYY-MM-DD for daily aggregation
);

-- Template Management
CREATE TABLE templates (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    category TEXT NOT NULL,
    description TEXT,
    template_content TEXT NOT NULL,
    variables TEXT, -- JSON array of template variables
    is_builtin BOOLEAN DEFAULT 0,
    created_by TEXT,
    created_at INTEGER DEFAULT (strftime('%s', 'now'))
);

-- Task State Management
CREATE TABLE tasks (
    id TEXT PRIMARY KEY,
    session_id TEXT NOT NULL,
    status TEXT NOT NULL, -- created, running, completed, failed, rolled_back
    state_data TEXT, -- JSON blob of task state
    checkpoints TEXT, -- JSON array of checkpoint states
    created_at INTEGER DEFAULT (strftime('%s', 'now')),
    updated_at INTEGER DEFAULT (strftime('%s', 'now'))
);

-- Budget Tracking
CREATE TABLE budgets (
    id TEXT PRIMARY KEY,
    user_id TEXT NOT NULL,
    task_id TEXT,
    total_tokens INTEGER DEFAULT 0,
    total_cost_cents INTEGER DEFAULT 0,
    action_count INTEGER DEFAULT 0,
    loop_detection TEXT, -- JSON array of recent action fingerprints
    created_at INTEGER DEFAULT (strftime('%s', 'now')),
    expires_at INTEGER
);
```

### License Integration Architecture
```typescript
// License validation middleware
interface LicenseMiddleware {
  validateLicense(licenseKey: string): Promise<License>;
  checkFeatureAccess(userId: string, feature: string): Promise<boolean>;
  trackUsage(userId: string, tool: string, usage: UsageData): Promise<void>;
  enforceLimits(userId: string, tool: string): Promise<boolean>;
}

// MCP Server with license gating
class AgentOSServer {
  private licenseMiddleware: LicenseMiddleware;
  
  async initialize() {
    // Free tools always available
    this.server.tool("format_convert", formatConvertHandler);
    this.server.tool("kv_store", kvStoreHandler);
    
    // Paid tools require license validation
    this.server.tool("task_ledger", withLicenseCheck(taskLedgerHandler));
    this.server.tool("budget_guard", withLicenseCheck(budgetGuardHandler));
    this.server.tool("template_engine", withLicenseCheck(templateEngineHandler));
  }
}

// License check decorator
function withLicenseCheck(handler: ToolHandler) {
  return async (args: ToolArgs) => {
    const license = await validateLicense(args.licenseKey);
    if (!license.hasFeature(args.toolName)) {
      throw new LicenseError("Feature requires Pro license");
    }
    return await handler(args);
  };
}
```
