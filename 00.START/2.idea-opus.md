### 1.2 — Tool #1: `kv_store` (Key-Value Memory)

> **Why this first**: Every agent needs persistent memory between sessions. This is the simplest tool to build and the one with the broadest utility.

| Tool | Description |
|---|---|
| `kv_set` | Store a key-value pair with optional TTL and namespace |
| `kv_get` | Retrieve a value by key |
| `kv_delete` | Remove a key |
| `kv_list` | List keys in a namespace with optional prefix filter |

**Implementation pattern** (repeat for all tools):

```typescript
import { z } from "zod";
import type { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { db } from "../db.js";

export function registerKvTools(server: McpServer) {
  server.tool(
    "kv_set",
    "Store a key-value pair in persistent local memory. " +
    "Supports namespaces for isolation and optional TTL for auto-expiry.",
    {
      key: z.string().describe("The key to store"),
      value: z.string().describe("The value to store (string or JSON)"),
      namespace: z.string().optional().default("default")
        .describe("Namespace for key isolation"),
      ttl_seconds: z.number().optional()
        .describe("Time-to-live in seconds. Omit for permanent storage."),
    },
    async ({ key, value, namespace, ttl_seconds }) => {
      try {
        const ttl = ttl_seconds 
          ? Math.floor(Date.now() / 1000) + ttl_seconds 
          : null;
        db.prepare(`
          INSERT OR REPLACE INTO kv (namespace, key, value, ttl, created_at)
          VALUES (?, ?, ?, ?, unixepoch())
        `).run(namespace, key, value, ttl);
        return {
          content: [{ 
            type: "text", 
            text: `Stored "${key}" in namespace "${namespace}"` +
              (ttl_seconds ? ` (expires in ${ttl_seconds}s)` : "") 
          }],
        };
      } catch (err) {
        return {
          content: [{ type: "text", text: `Error: ${(err as Error).message}` }],
          isError: true,
        };
      }
    }
  );
  
  // ... kv_get, kv_delete, kv_list follow same pattern
}
```

AVAILABLE CAPABILITIES:
- kv_set/kv_get/kv_delete/kv_list: Persistent key-value memory across sessions

  console.error("Fatal:", err);

const LICENSE_KEY = process.env.AGENTOS_LICENSE_KEY;
const PRO_TOOLS = new Set([
  "task_create", "task_checkpoint", /* ... */
]);

export function isProUnlocked(): boolean {
  if (!LICENSE_KEY) return false;
  // Simple validation: check key format + optional API call
  return LICENSE_KEY.startsWith("aos_") && LICENSE_KEY.length === 32;

export function gatePro(toolName: string): string | null {
  if (!PRO_TOOLS.has(toolName)) return null; // Free tool
  if (isProUnlocked()) return null; // Licensed

    error: "pro_required",

### 2.5.2 — Advanced License Management

```typescript
// src/license-advanced.ts
export interface License {
  id: string;
  user_id: string;
  tier: 'free' | 'pro' | 'team' | 'enterprise';
  features: string[];
  workspace_limits: number;
  expires_at?: number;
  created_at: number;
}

export class LicenseManager {
  private cache = new Map<string, License>();
  private cache_ttl = 300000; // 5 minutes
  
  async validateLicense(key: string): Promise<License | null> {
    // Check cache first
    if (this.cache.has(key)) {
      const cached = this.cache.get(key)!;
      if (Date.now() - cached.created_at < this.cache_ttl) {
        return cached;
      }
    }
    
    // Validate with marketplace or local
    const license = await this.validateWithMarketplace(key) || 
                    await this.validateLocal(key);
    
    if (license) {
      this.cache.set(key, license);
    }
    
    return license;
  }
  
  async validateWithMarketplace(key: string): Promise<License | null> {
    try {
      const response = await fetch('https://api.mcp-marketplace.io/validate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ license_key: key, product: 'agentos-mcp' })
      });
      
      if (response.ok) {
        const data = await response.json();
        return {
          id: data.license_id,
          user_id: data.user_id,
          tier: data.tier,
          features: data.features,
          workspace_limits: data.workspace_limits,
          expires_at: data.expires_at,
          created_at: Date.now()
        };
      }
    } catch (error) {
      console.error('Marketplace validation failed:', error);
    }
    
    return null;
  }
}
```
